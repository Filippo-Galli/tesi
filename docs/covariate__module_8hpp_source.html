109 inline double compute_log_marginal_likelihood_NN(const ClusterStats &stats) const __attribute__((hot)) {
110 if (stats.n == 0) {
111 return 0.0;
112 }
113
114 const double n_dbl = static_cast<double>(stats.n);
115 const double inv_n = 1.0 / n_dbl;
116 const double xbar = stats.sum * inv_n;
117
118 // Centered sum of squares: SS = Σ(x_i - x̄)²
119 const double ss = stats.sumsq - n_dbl * xbar * xbar;
120
121 // v + n_j B term from integrating μ_j ~ N(m, B)
122 const double v_plus_nB = covariates_data.v + n_dbl * covariates_data.B;
123
124 // Mean deviation from prior mean
125 const double dev = xbar - covariates_data.m;
126
127 // Log of posterior variance: log(τ_j) = log(B) + log(v) - log(v + n_j B)
128 const double log_tau_j = log_B + log_v - std::log(v_plus_nB);
129
130 const double temp_log_v_plus_nB = log_v_plus_nB[stats.n];
131
132 // Compute log marginal likelihood using posterior variance form:
133 // log q(x_j) = -n_j/2 log(2π) - n_j/2 log(v) - 1/2 log(B) + 1/2 log(τ_j)
134 // - SS/(2v) - n_j(x̄_j - m)² / (2(v + n_j B))
135 return n_dbl * const_term - 0.5 * n_dbl * log_v - 0.5 * log_B + 0.5 * log_tau_j -
136 0.5 * (ss / covariates_data.v + n_dbl * dev * dev / v_plus_nB);
137 }

155 inline double compute_log_marginal_likelihood_NNIG(const ClusterStats &stats) const __attribute__((hot)) {
156 if (stats.n == 0) {
157 return 0.0;
158 }
159
160 const double n_dbl = static_cast<double>(stats.n);
161 const double inv_n = 1.0 / n_dbl;
162 const double xbar = stats.sum * inv_n;
163
164 // Centered sum of squares: SS = Σ(xᵢ - x̄)²
165 const double ss = stats.sumsq - n_dbl * xbar * xbar;
166
167 // Posterior parameters
168 // Prior: v ~ IG(ν, S0) and μ|v ~ N(m, B v) with B interpreted as a variance multiplier.
169 const double nu_n = covariates_data.nu + 0.5 * n_dbl;
170
171 // Deviation from prior mean
172 const double dev = xbar - covariates_data.m;
173
174 // Posterior scale parameter:
175 // S_n = S₀ + SS/2 + n/(2(1+nB)) (x̄-m)²
176 const double one_plus_nB = 1.0 + n_dbl * covariates_data.B;
177 const double S_n = covariates_data.S0 + 0.5 * ss + 0.5 * (n_dbl / one_plus_nB) * dev * dev;
178
179 const double temp_lgamma_nu_n = lgamma_nu_n[stats.n];
180
181 // log g(S) = log Γ(ν + n/2) - log Γ(ν) - n/2 log(2π)
182 // - 1/2 log(1+nB) + ν log(S₀) - (ν + n/2) log(S_n)
183 return temp_lgamma_nu_n - lgamma_nu + n_dbl * const_term - 0.5 * std::log(one_plus_nB) + nu_logS0 -
184 nu_n * std::log(S_n);
185 }

194 const double Bv = covariates_data.B * covariates_data.v;

196 const double log_B = std::log(covariates_data.B);

198
