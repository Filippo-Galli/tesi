112 inline double compute_log_marginal_likelihood_NN(const ClusterStats &stats) const __attribute__((hot)) {
113 if (stats.n == 0) {
114 return 0.0;
115 }
116
117 const double n_dbl = static_cast<double>(stats.n);
118 const double inv_n = 1.0 / n_dbl;
119 const double xbar = stats.sum * inv_n;
120
121 // Centered sum of squares: SS = Σ(xᵢ - x̄)²
122 const double ss = stats.sumsq - n_dbl * xbar * xbar;
123
124 // Posterior variance parameter: B + nv (NOT v + nB!)
125 const double B_nv = covariates_data.B + n_dbl * covariates_data.v;
126
127 // Deviation from prior mean
128 const double dev = xbar - covariates_data.m;
129
130 // log g(S) = -n/2 log(2π) - n/2 log(v) + 1/2 log(B) - 1/2 log(B + nv)
131 // - SS/(2v) - n(x̄ - m)²/(2(B + nv))
132 return n_dbl * (const_term - 0.5 * log_v) + 0.5 * (log_B - std::log(B_nv)) -
133 0.5 * (ss / covariates_data.v + n_dbl * dev * dev / B_nv);
134 }
